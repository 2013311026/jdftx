315 void eblas_mul_sub(size_t iMin, size_t iMax, const Tx* X, const int incX, Ty* Y, const int incY)
316 { for(size_t i=iMin; i<iMax; i++) Y[incY*i] *= X[incX*i];
317 }

319 template<typename Ty, typename Tx>
320 void eblas_mul(const int N, const Tx* X, const int incX, Ty* Y, const int incY)
321 { if(incY==0) die("incY cannot be = 0")
322  threadLaunch((N<100000) ? 1 : 0, //force single threaded for small problem sizes
323  eblas_mul_sub<Ty,Tx>, N, X, incX, Y, incY);
324 }
