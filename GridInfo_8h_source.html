130  //Indexing utilities (inlined for efficiency)
131  inline vector3<int> wrapGcoords(const vector3<int> iG) const
132  { vector3<int> iGwrapped = iG;
133  for(int k=0; k<3; k++)
134  if(iGwrapped[k]<0)
135  iGwrapped[k] += S[k];
136  return iGwrapped;
137  }
138  inline int fullRindex(const vector3<int> iR) const
139  { return iR[2] + S[2]*(iR[1] + S[1]*iR[0]);
140  }
141  inline int fullGindex(const vector3<int> iG) const
142  { return fullRindex(wrapGcoords(iG));
143  }
144  inline int halfGindex(const vector3<int> iG) const
145  { vector3<int> iGwrapped = wrapGcoords(iG);
146  return iGwrapped[2] + (S[2]/2+1)*(iGwrapped[1] + S[1]*iGwrapped[0]);
147  }
